Firestore rules for v1.3.5

These rules let anyone read events and signups, let anyone create a signup, and only let the signed in event owner manage the event or signup list.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /openMicEvents/{eventId} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.hostUid == request.auth.uid;
      allow update, delete: if request.auth != null
        && resource.data.hostUid == request.auth.uid;

      match /signups/{signupId} {
        allow read: if true;
        allow create: if true;
        allow update, delete: if request.auth != null
          && get(/databases/$(database)/documents/openMicEvents/$(eventId)).data.hostUid == request.auth.uid;
      }
    }
  }
}
